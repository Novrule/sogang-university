CREATE DATABASE project;

USE project;

CREATE TABLE district_info (district VARCHAR(20) PRIMARY KEY, school_district VARCHAR(20) NOT NULL);
CREATE TABLE properties (property_id VARCHAR(20) PRIMARY KEY, property_name VARCHAR(20), property_type VARCHAR(20), sale_type VARCHAR(20), sale_status BIT NOT NULL DEFAULT 0, construction_date DATETIME, price INT NOT NULL, area INT, district VARCHAR(20), street_name VARCHAR(20), street_number VARCHAR(20), living_room_count INT CHECK (living_room_count >= 0), bed_room_count INT CHECK (bed_room_count >= 0), bath_room_count INT CHECK (bath_room_count >= 0), FOREIGN KEY (district) REFERENCES district_info(district));
CREATE TABLE agents (agent_id VARCHAR(20) PRIMARY KEY, agent_name VARCHAR(20) NOT NULL);
CREATE TABLE manage_property (agent_id VARCHAR(20) NOT NULL, property_id VARCHAR(20) NOT NULL, agent_type BIT NOT NULL DEFAULT 0, PRIMARY KEY (agent_id, property_id), FOREIGN KEY (agent_id) REFERENCES agents(agent_id), FOREIGN KEY (property_id) REFERENCES properties(property_id));
CREATE TABLE agent_info (agent_id VARCHAR(20) NOT NULL, phone_number VARCHAR(20) NOT NULL, PRIMARY KEY (agent_id, phone_number), FOREIGN KEY (agent_id) REFERENCES agents(agent_id));
CREATE TABLE buyers (buyer_id VARCHAR(20) PRIMARY KEY, buyer_name VARCHAR(20) NOT NULL);
CREATE TABLE buy_property (buyer_id VARCHAR(20) NOT NULL, property_id VARCHAR(20) NOT NULL, buy_date DATETIME NOT NULL, PRIMARY KEY (buyer_id, property_id), FOREIGN KEY (buyer_id) REFERENCES buyers(buyer_id), FOREIGN KEY (property_id) REFERENCES properties(property_id));
CREATE TABLE buyer_info (buyer_id VARCHAR(20) NOT NULL, phone_number VARCHAR(20) NOT NULL, PRIMARY KEY (buyer_id, phone_number), FOREIGN KEY (buyer_id) REFERENCES buyers(buyer_id));
CREATE TABLE sellers (seller_id VARCHAR(20) PRIMARY KEY, seller_name VARCHAR(20) NOT NULL);
CREATE TABLE sell_property (seller_id VARCHAR(20) NOT NULL, property_id VARCHAR(20) NOT NULL, PRIMARY KEY (seller_id, property_id), FOREIGN KEY (seller_id) REFERENCES sellers(seller_id), FOREIGN KEY (property_id) REFERENCES properties(property_id));
CREATE TABLE seller_info (seller_id VARCHAR(20) NOT NULL, phone_number VARCHAR(20) NOT NULL, PRIMARY KEY (seller_id, phone_number), FOREIGN KEY (seller_id) REFERENCES sellers(seller_id));
CREATE TABLE floor_plans (property_id VARCHAR(20) NOT NULL, floor_plan_name VARCHAR(20) NOT NULL, floor_plan_file BLOB NOT NULL, PRIMARY KEY (property_id, floor_plan_name), FOREIGN KEY (property_id) REFERENCES properties(property_id));
CREATE TABLE photos (property_id VARCHAR(20) NOT NULL, photo_name VARCHAR(20) NOT NULL, photo_file BLOB NOT NULL, PRIMARY KEY (property_id, photo_name), FOREIGN KEY (property_id) REFERENCES properties(property_id));
CREATE TABLE property_bonus (property_id VARCHAR(20) NOT NULL, bonus VARCHAR(20) NOT NULL, PRIMARY KEY (property_id, bonus), FOREIGN KEY (property_id) REFERENCES properties(property_id));
